!|Execute|create or replace force type person_type as object(id number, name varchar2(50 char))|
!|Execute|create or replace force type person_type_tab as varray(20) of person_type|

!|Execute|create table persons(reg_id integer, member person_type)|

!|Execute|!-
create procedure store_persons(persons person_type_tab) as
  id integer := 1;
begin
  for i in persons.first .. persons.last loop
    insert into persons(reg_id, member) values (id, persons(i));
    id := id + 1;
  end loop;
end;
-!|
